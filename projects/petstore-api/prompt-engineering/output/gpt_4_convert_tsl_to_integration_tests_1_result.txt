import io.restassured.RestAssured;
import io.restassured.http.ContentType;
import io.restassured.response.Response;
import org.junit.jupiter.api.Test;

import static io.restassured.RestAssured.given;
import static org.hamcrest.Matchers.equalTo;
import static org.hamcrest.Matchers.notNullValue;

public class PetEndpointTests {

    static {
        RestAssured.baseURI = "https://petstore3.swagger.io/api/v3";
    }

    private String createPet(String name, String status) {
        String petPayload = """
            {
                "name": "%s",
                "status": "%s",
                "photoUrls": ["https://example.com/photo.jpg"]
            }
        """.formatted(name, status);

        Response response = given()
                .contentType(ContentType.JSON)
                .body(petPayload)
                .post("/pet");
        response.then().statusCode(200).and().body("id", notNullValue());

        return response.jsonPath().getString("id");
    }

    private void deletePet(String petId) {
        given()
                .delete("/pet/" + petId)
                .then()
                .statusCode(200);
    }

    @Test
    public void testUpdatePet_validInputs() {
        // Arrange
        String petId = createPet("doggie", "available");
        String updatePayload = """
            {
                "id": %s,
                "name": "updatedDoggie",
                "status": "sold",
                "photoUrls": ["https://example.com/updated_photo.jpg"]
            }
        """.formatted(petId);

        // Act
        Response response = given()
                .contentType(ContentType.JSON)
                .body(updatePayload)
                .put("/pet");

        // Assert
        response.then()
                .statusCode(200)
                .body("id", equalTo(Integer.parseInt(petId)))
                .body("name", equalTo("updatedDoggie"))
                .body("status", equalTo("sold"));

        // Cleanup
        deletePet(petId);
    }

    @Test
    public void testUpdatePet_invalidId() {

            {
