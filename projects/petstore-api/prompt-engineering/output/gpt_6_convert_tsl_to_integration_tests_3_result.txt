import io.restassured.RestAssured;
import io.restassured.http.ContentType;
import io.restassured.response.Response;
import org.junit.jupiter.api.Test;

import java.util.UUID;

import static io.restassured.RestAssured.given;
import static org.hamcrest.Matchers.equalTo;
import static org.hamcrest.Matchers.notNullValue;

public class UserEndpointTests {

    static {
        RestAssured.baseURI = "https://petstore3.swagger.io/api/v3";
    }

    private String createUser(String username, String password) {
        String userPayload = """
            {
                "username": "%s",
                "firstName": "John",
                "lastName": "Doe",
                "email": "john.doe@example.com",
                "password": "%s",
                "phone": "123456789",
                "userStatus": 1
            }
        """.formatted(username, password);

        Response response = given()
                .contentType(ContentType.JSON)
                .body(userPayload)
                .post("/user");
        response.then().statusCode(200);

        return username;
    }

    private void deleteUser(String username) {
        given()
                .delete("/user/" + username)
                .then()
                .statusCode(200);
    }

    @Test
    public void testCreateUser_validInputs() {
        // Arrange
        String uniqueUsername = "user_" + UUID.randomUUID();
        String password = "validPassword";

        String userPayload = """
            {
                "username": "%s",
                "firstName": "John",

                "email": "john.doe@example.com",
                "password": "%s",
                "phone": "123456789",

            }
        """.formatted(uniqueUsername, password);

        // Act
        Response response = given()
                .contentType(ContentType.JSON)
                .body(userPayload)
                .post("/user");

        // Assert
        response.then().statusCode(200);

        // Cleanup
        deleteUser(uniqueUsername);
    }

    public void testGetUser_validUsername() {
        // Arrange
        String uniqueUsername = "user_" + UUID.randomUUID();

        createUser(uniqueUsername, password);

        Response response = given()
                .get("/user/" + uniqueUsername);

        // Assert
        response.then()
