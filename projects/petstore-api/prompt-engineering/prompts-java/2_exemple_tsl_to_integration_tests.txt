import io.restassured.RestAssured;
import io.restassured.http.ContentType;
import io.restassured.response.Response;
import org.junit.jupiter.api.*;
import static io.restassured.RestAssured.given;
import static org.hamcrest.Matchers.*;

@TestMethodOrder(MethodOrderer.OrderAnnotation.class)

    @BeforeAll
    public static void setup() {
        RestAssured.baseURI = "http://localhost:8080";
    }

    private String createUser(String nickname, String password) {
        given()
            .contentType(ContentType.JSON)
            .body("{\"nickname\":\"" + nickname + "\",\"password\":\"" + password + "\"}")
        .when()
            .post("/users")
        .then()
            .statusCode(anyOf(is(200), is(400)));
        return nickname;
    }

    private String getToken(String nickname, String password) {
        return given()
            .contentType(ContentType.JSON)
            .body("{\"nickname\":\"" + nickname + "\",\"password\":\"" + password + "\"}")
        .when()
            .post("/users/token")
        .then()
            .extract()
            .path("token");
    }

    private Response createPerson(String token, String name, Object age) {
        return given()
            .contentType(ContentType.JSON)
            .header("Authorization", token != null ? "Bearer " + token : null)
            .body("{\"name\": " + (name == null ? null : "\"" + name + "\"") + ", \"age\": " + age + "}")
        .when()
            .post("/people");
    }

    private Response getPerson(String token, int id) {
        return given()
            .header("Authorization", token != null ? "Bearer " + token : null)
        .when()
            .get("/people/" + id);
    }
