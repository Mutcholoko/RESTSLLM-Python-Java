Agora vou te ensinar boas práticas na conversão de um TSL para testes de integração com Python (PyTest).

Dado o TSL fornecido anteriormente, uma conversão para testes de integração com Python (PyTest) deveria gerar:

```python
import pytest
import requests

BASE_URL = "http://localhost:8080"

@pytest.fixture(scope="session")
def base_url():
    return BASE_URL

def create_user(nickname, password):
    body = {"nickname": nickname, "password": password}
    return requests.post(f"{BASE_URL}/users", json=body)

def create_token(nickname, password):
    body = {"nickname": nickname, "password": password}
    return requests.post(f"{BASE_URL}/users/token", json=body)

def create_person(token, name, age):
    headers = {"Authorization": f"Bearer {token}"} if token else {}
    body = {"name": name, "age": age}
    return requests.post(f"{BASE_URL}/people", json=body, headers=headers)

def get_person(token, person_id):
    headers = {"Authorization": f"Bearer {token}"} if token else {}
    return requests.get(f"{BASE_URL}/people/{person_id}", headers=headers)

# ==================== USERS ====================

def test_TC001_create_user_valid_data_returns_ok():
    resp = create_user("validNickname1", "ValidPass1")
    assert resp.status_code == 200

def test_TC002_create_user_nickname_already_exists_returns_bad_request():
    nickname = "existingNick"
    password = "ValidPass1"
    create_user(nickname, password)
    resp = create_user(nickname, password)
    assert resp.status_code == 400

@pytest.mark.parametrize("nickname,password,status", [
    ("invalid!Nick", "ValidPass1", 400),
    (None, "ValidPass1", 400),
    ("", "ValidPass1", 400),
    ("invalidNi", "ValidPass1", 400),
    ("validNick1abcabcabcdabcabcabcd123", "ValidPass1", 400),
    ("validNick0", "ValidPass1", 200),
    ("validNick1abcabcabcdabcabcabcd12", "ValidPass1", 200),
])
def test_TC003_to_TC009_create_user_nickname_validations(nickname, password, status):
    body = {"nickname": nickname, "password": password}
    resp = requests.post(f"{BASE_URL}/users", json=body)
    assert resp.status_code == status

@pytest.mark.parametrize("nickname,password,status", [
    ("validNick1", None, 400),
    ("validNick2", "", 400),
    ("validNick3", "Abcd1", 400),
    ("validNick4", "Ab1Abc2bc2bc2bc2bc2bc", 400),
    ("validNick5", "Abcde1", 200),
    ("validNick6", "Ab1Abc2bc2bc2bc2bc2b", 200),
    ("validNick7", "validpass1", 400),
    ("validNick8", "VALIDPASS1", 400),
    ("validNick9", "ValidPassX", 400),
])
def test_TC010_to_TC018_create_user_password_validations(nickname, password, status):
    body = {"nickname": nickname, "password": password}
    resp = requests.post(f"{BASE_URL}/users", json=body)
    assert resp.status_code == status

def test_TC019_authenticate_user_valid_data_returns_ok():
    nickname = "validNick10"
    password = "ValidPass1"
    create_user(nickname, password)
    resp = create_token(nickname, password)
    assert resp.status_code == 200
    token = resp.json().get("token", "")
    assert isinstance(token, str)
    assert token != ""

def test_TC020_authenticate_user_invalid_password_returns_bad_request():
    nickname = "validNick11"
    password = "ValidPass1"
    create_user(nickname, password)
    resp = create_token(nickname, "WrongPass1")
    assert resp.status_code == 400

def test_TC021_authenticate_user_invalid_nickname_returns_bad_request():
    nickname = "validNick12"
    password = "ValidPass1"
    create_user(nickname, password)
    resp = create_token("validNickname12", password)
    assert resp.status_code == 400

# ==================== PEOPLE ====================

def test_TC022_create_person_valid_data_returns_created():
    nickname, password = "validNick13", "ValidPass1"
    create_user(nickname, password)
    token = create_token(nickname, password).json()["token"]
    resp = create_person(token, "John Doe", 30)
    body = resp.json()
    assert resp.status_code == 201
    assert body["id"] > 0
    assert body["name"] == "John Doe"
    assert body["age"] == 30

@pytest.mark.parametrize("name,age,status", [
    (None, 30, 400),
    ("", 30, 400),
    ("", 30, 400),
    ("John Doe abcabcabcgabcabcabcgabcabcabcgabcabcabcgabcabcabcgabcabcabcgabcabcabcgabcabcabcgabcabcabcgabcabcabcgabcabcabcgabcabcabcg", 30, 400),
])
def test_TC023_to_TC026_create_person_name_validations(name, age, status):
    nickname, password = "validNick" + str(status), "ValidPass1"
    create_user(nickname, password)
    token = create_token(nickname, password).json()["token"]
    resp = create_person(token, name, age)
    assert resp.status_code == status

@pytest.mark.parametrize("name,age,status", [
    ("J", 30, 201),
    ("John Doeabcabcabcgabcabcabcgabcabcabcgabcabcabcgabcabcabcgabcabcabcgabcabcabcgabcabcabcgabcabcabcgabcabcabcgabcabcabcgabcabcabcg", 30, 201),
    ("John Doe", None, 400),
    ("John Doe", -1, 400),
    ("John Doe", 151, 400),
    ("John Doe", 0, 201),
    ("John Doe", 150, 201),
])
def test_TC027_to_TC033_create_person_age_validations(name, age, status):
    nickname, password = f"validNick{age if age else 'null'}", "ValidPass1"
    create_user(nickname, password)
    token = create_token(nickname, password).json()["token"]
    resp = create_person(token, name, age)
    assert resp.status_code == status

def test_TC034_create_person_invalid_token_returns_unauthorized():
    resp = create_person("invalidtoken", "John Doe", 30)
    assert resp.status_code == 401

def test_TC035_create_person_without_token_returns_unauthorized():
    resp = create_person(None, "John Doe", 30)
    assert resp.status_code == 401

def test_TC036_get_person_valid_data_returns_ok():
    nickname, password = "validNick25", "ValidPass1"
    create_user(nickname, password)
    token = create_token(nickname, password).json()["token"]
    create_resp = create_person(token, "John Doe", 30)
    person_id = create_resp.json()["id"]
    resp = get_person(token, person_id)
    body = resp.json()
    assert resp.status_code == 200
    assert body["id"] == person_id
    assert body["name"] == "John Doe"
    assert body["age"] == 30

def test_TC037_get_person_not_exists_returns_not_found():
    nickname, password = "validNick26", "ValidPass1"
    create_user(nickname, password)
    token = create_token(nickname, password).json()["token"]
    resp = get_person(token, 9999999)
    assert resp.status_code == 404

def test_TC038_get_person_invalid_token_returns_unauthorized():
    resp = get_person("invalidtoken", 1)
    assert resp.status_code == 401

def test_TC039_get_person_without_token_returns_unauthorized():
    resp = get_person(None, 1)
    assert resp.status_code == 401

def test_TC040_get_person_token_from_other_user_returns_not_found():
    token1_user = "validNick27"
    token2_user = "validNick28"
    password = "ValidPass1"
    create_user(token1_user, password)
    token1 = create_token(token1_user, password).json()["token"]
    person_id = create_person(token1, "John Doe", 30).json()["id"]
    create_user(token2_user, password)
    token2 = create_token(token2_user, password).json()["token"]
    resp = get_person(token2, person_id)
    assert resp.status_code == 404

# ==================== APP ====================

def test_TC041_update_app_logo_valid_content_type_returns_ok():
    files = {"file": ("somefile.jpg", b"123456", "image/jpeg")}
    resp = requests.put(f"{BASE_URL}/app/logo", files=files)
    assert resp.status_code == 200

def test_TC042_update_app_logo_invalid_content_type_returns_unsupported_media_type():
    body = {"file": "somefile.jpg"}
    resp = requests.put(f"{BASE_URL}/app/logo", json=body)
    assert resp.status_code == 415
```
